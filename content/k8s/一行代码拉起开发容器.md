[taxonomies]

```shell
# nats
docker run --restart=always -d  --name nats \
 -p 4222:4222 \
 -p 8222:8222 \
 nats --http_port 8222
# clickhouse 
docker run -d --name ch --restart=always -p 8123:8123 -p 9000:9000 -e CLICKHOUSE_ADMIN_PASSWORD=ch123456 bitnami/clickhouse:25
# etcd
docker run --restart=always -d --name etcd \
    --publish 2379:2379 \
    --publish 2380:2380 \
    --env ALLOW_NONE_AUTHENTICATION=yes \
    --env ETCD_ADVERTISE_CLIENT_URLS=http://127.0.0.1:2379 \
    bitnami/etcd:latest
# mongo
# -e ALLOW_EMPTY_PASSWORD=yes 密码为空
docker run --restart=always -d --name mongo \
  --publish 27017:27017 \
  -e MONGO_INITDB_ROOT_USERNAME=root \
  -e MONGO_INITDB_ROOT_PASSWORD=root123456 \
  mongo
```
